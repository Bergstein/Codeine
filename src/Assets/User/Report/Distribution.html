function drawChart(data) { var data = google.visualization.arrayToDataTable(data); var options = { title: 'Распределение по «ID»' }; var chart = new google.visualization.PieChart(document.getElementById('chart_div')); chart.draw(data, options); } $.ajax({ url: '/report/User/Distribution/ID.json', dataType: 'json', success: function (data) {drawChart(data)} })
